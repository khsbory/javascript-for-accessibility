// 실시간 변경을 감지할 요소
let target = document.querySelector("#vipContent > div:nth-child(11)");

// 옵저버 인스턴스 생성
let observer = new MutationObserver((mutations) => {
    // 노드가 변경 됐을 때의 작업, 여기에 none 혹은 block시에 대한 이벤트를 작성해주시면 됩니다
    alert('해당 요소의 변경이 감지되었습니다');
})

// 옵션 설정
let option = {
    attributes: true, // 대상 노드의 속성에 대한 변화를 감시
    childList: true, // 대상 노드의 하위 요소의 추가 및 제거를 감시
    characterData: true // 대상 노드의 데이터에 대한 변화를 감시
};

// 실행
observer.observe(target, option);
